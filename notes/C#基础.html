2. 控制结构

2.1 条件语句

// if-else
if (age <= 18)
{
    Console.WriteLine("成年人");
}
else
{
    Console.WriteLine("未成年人");
}

// switch-case
switch (grade)
{
    case 'A':
        Console.WriteLine("优秀");
        break;
    case 'B':
        Console.WriteLine("良好");
        break;
    default:
        Console.WriteLine("其他");
        break;
}

2.2 循环结构

// for循环
for (int i = 0; i < 10; i++)
{
    Console.WriteLine(i);
}

// while循环
int j = 0;
while (j < 10)
{
    Console.WriteLine(j);
    j++;
}

// foreach循环
string[] colors = { "红", "绿", "蓝" };
foreach (string color in colors)
{
    Console.WriteLine(color);
}

3. 面向对象编程

3.1 类与对象

public class Person
{
    // 字段
    private string name;
    
    // 属性
    public string Name
    {
        get { return name; }
        set { name = value; }
    }
    
    // 方法
    public void SayHello()
    {
        Console.WriteLine($"你好，我是{Name}");
    }
    
    // 构造函数
    public Person(string name)
    {
        Name = name;
    }
}

// 使用类
Person person = new Person("李四");
person.SayHello();

3.2 继承与多态

public class Student : Person
{
    public string StudentId { get; set; }
    
    public Student(string name, string studentId) : base(name)
    {
        StudentId = studentId;
    }
    
    // 方法重写
    public override void SayHello()
    {
        Console.WriteLine($"你好，我是学生{Name}, 学号:{StudentId}");
    }
}
